    <script>
        document.getElementById('report-form').addEventListener('submit', function (event) {
            event.preventDefault();

            var form = event.target;
            var statusContainer = document.getElementById("status");
            var resultadoContainer = document.getElementById("resultado");
            var erroContainer = document.getElementById("erro");

            
            statusContainer.innerHTML = 'Script está sendo executado...';
            statusContainer.style.display = 'block';
            resultadoContainer.style.display = 'none';
            erroContainer.style.display = 'none';

            
            fetch(form.action, {
                method: 'POST',
                body: new FormData(form)
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    
                    const intervalId = setInterval(() => {
                        fetch('/status')
                            .then(response => response.json())
                            .then(data => {
                                if (data.etapa) {
                                    statusContainer.innerHTML = data.etapa;
                                    statusContainer.style.display = 'block';
                                } else {
                                    statusContainer.style.display = 'none';
                                }

                                if (data.excel_final) {
                                    resultadoContainer.innerHTML = data.excel_final;
                                    resultadoContainer.style.display = 'block';
                                    statusContainer.style.display = 'none';
                                    clearInterval(intervalId);  
                                }

                                if (data.erro) {
                                    erroContainer.innerHTML = data.erro;
                                    erroContainer.style.display = 'block';
                                    statusContainer.style.display = 'none';
                                    clearInterval(intervalId);  
                                }
                            })
                            .catch(error => {
                                erroContainer.innerHTML = 'Erro ao obter status do servidor';
                                erroContainer.style.display = 'block';
                                statusContainer.style.display = 'none';
                                clearInterval(intervalId);  
                            });
                    }, 3000);  
                } else {
                    erroContainer.innerHTML = data.message;
                    erroContainer.style.display = 'block';
                    statusContainer.style.display = 'none';
                }
            })
            .catch(error => {
                erroContainer.innerHTML = 'Erro ao enviar o formulário';
                erroContainer.style.display = 'block';
                statusContainer.style.display = 'none';
            });
        });
    </script>
